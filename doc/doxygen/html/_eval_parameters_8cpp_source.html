76 void NOMAD::EvalParameters::checkAndComply(const std::shared_ptr<NOMAD::RunParameters>& runParams,
77  const std::shared_ptr<NOMAD::PbParameters>& pbParams)
78 {
79  checkInfo();
80 
81  if (!toBeChecked())
82  {
83  // Early out
84  return;
85  }
86 
87  // Update BB_EXE if it was set by user:
88  // - Set full path
89  // - Remove '$' indicating a global call (e.g. python, perl)
90  // - Verify file is executable
91  if (isSetByUser("BB_EXE"))
92  {
93  auto bbExe = getAttributeValueProtected<std::string>("BB_EXE", false);
94  /*
95  // Ignore; the isSetByUser() is somehow sticky, so this causes issues in unit tests.
96  if (bbExe.empty())
97  {
98  throw NOMAD::Exception(__FILE__, __LINE__, "BB_EXE is not defined");
99  }
100  */
101 
102  bool localExe = true;
103  auto problemDir = runParams->getAttributeValue<std::string>("PROBLEM_DIR");
104 
105  if ('$' == bbExe[0])
106  {
107  // When the '$' character is put in first
108  // position of a string, it is considered
109  // as global and no path will be added.
110  localExe = false;
111  }
112 
113  // Convert arguments; add path as needed.
114  auto bbExeAsArray = NOMAD::ArrayOfString(bbExe);
115  bbExe.clear();
116  for (size_t i = 0; i < bbExeAsArray.size(); i++)
117  {
118  std::string word = bbExeAsArray[i];
119  if (i > 0)
120  {
121  bbExe += " ";
122  }
123 
124  if ('$' == word[0])
125  {
126  bbExe += word.substr(1, word.size());
127  }
128  else
129  {
130  // word is relative to problem directory.
131  completeFileName(word, problemDir);
132  bbExe += word;
133  }
134  }
135 
136  setAttributeValue("BB_EXE", bbExe);
137  bbExeAsArray = NOMAD::ArrayOfString(bbExe);
138 
139  if (localExe && !bbExe.empty() && !checkExeFile(bbExeAsArray[0]))
140  {
141  throw NOMAD::Exception(__FILE__, __LINE__, "BB_EXE needs to be an executable file: " + bbExeAsArray[0]);
142  }
143  }
144 
145  /*----------------*/
146  /* BB_OUTPUT_TYPE */
147  /*----------------*/
148  // The default value is empty: set a single OBJ
149  auto bbOType = getAttributeValueProtected<NOMAD::BBOutputTypeList>("BB_OUTPUT_TYPE", false);
150  if ( bbOType.size() == 0 )
151  {
152  bbOType.push_back(NOMAD::BBOutputType::OBJ);
153  setAttributeValue("BB_OUTPUT_TYPE", bbOType );
154  }
155 
156  /*---------------------------*/
157  /* BB_EVAL_FORMAT (internal) */
158  /*---------------------------*/
159  size_t n = pbParams->getAttributeValue<size_t>("DIMENSION");
160  auto evalFormat = getAttributeValueProtected<NOMAD::ArrayOfDouble>("BB_EVAL_FORMAT",false);
161  if (!evalFormat.isDefined())
162  {
163  // Default precision is computed based on Epsilon.
164  int defaultPrec = static_cast<int>(-log10(NOMAD::Double::getEpsilon())) + 1;
165  evalFormat.reset(n, defaultPrec);
166  setAttributeValue("BB_EVAL_FORMAT", evalFormat);
167  }
168 
169  auto bbInputType = pbParams->getAttributeValue<NOMAD::BBInputTypeList>("BB_INPUT_TYPE");
170  // CONTINUOUS variables will be written with full precision.
171  // INTEGER, BOOLEAN, and eventually CATEGORICAL variables will be written
172  // as integers.
173  for (size_t i = 0; i < n; i++)
174  {
175  if (NOMAD::BBInputType::CONTINUOUS != bbInputType[i])
176  {
177  evalFormat[i] = -1;
178  }
179  }
180  setAttributeValue("BB_EVAL_FORMAT", evalFormat);
181 
182  _toBeChecked = false;
183 
184 }
185 // End checkAndComply()
