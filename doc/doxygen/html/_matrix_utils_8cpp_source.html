50 bool NOMAD::getDeterminant(double ** M,
51  double & det,
52  size_t n)
53 {
54  std::string error_msg;
55 
56  double d = 1 ;
57 
58  double ** LU = new double *[n];
59  for (size_t i = 0 ; i < n ; i++ )
60  {
61  LU[i]=new double [n];
62  for ( size_t j = 0; j < n ; j++ )
63  LU[i][j]=M[i][j];
64  }
65 
66  NOMAD::LU_decomposition ( error_msg, LU, static_cast<int>(n), d );

68

72  d*=LU[i][i];

74 
75

78  delete [] LU[i];
79  }

85  return true;

130 /*--------------------------------------------------------------*/
131 /* SVD decomposition */
132 /* inspired and recoded from an old numerical recipes version */
133 /*--------------------------------------------------------------*/
134 /* */
135 /* M = U . W . V' */
136 /* */
137 /* M ( m x n ) */
138 /* U ( m x n ) */
139 /* W ( n x n ) */
140 /* V ( n x n ) */
141 /* */
142 /* U.U' = U'.U = I if m = n */
143 /* U'.U = I if m > n */
144 /* */
145 /* V.V' = V'.V = I */
146 /* */
147 /* W diagonal */
148 /* */
149 /* M is given as first argument and becomes U */
150 /* W is given as a size-n vector */
151 /* V is given, not V' */
152 /* */
153 /*--------------------------------------------------------------*/
154 bool NOMAD::SVD_decomposition ( std::string & error_msg,
155  double ** M,
156  double * W,
157  double ** V,
158  int m,
159  int n,
160  int max_mpn ) // default = 1500
161 {
162  error_msg.clear();
163 
164  if ( max_mpn > 0 && m+n > max_mpn )
165  {
166  error_msg = "SVD_decomposition() error: m+n > " + NOMAD::itos ( max_mpn );
167  return false;
168  }
169 
170  double * rv1 = new double[n];
171  double scale = 0.0;
172  double g = 0.0;
173  double norm = 0.0;
174 
175  int nm1 = n - 1;
176 
177  bool flag;
178  int i, j, k, l = 0, its, jj, nm = 0;
179  double s, f, h, c, x, y, z, absf, absg, absh;
180 
181  const int NITER = 30;
182 
183  // Initialization W and V
184  for (i=0; i < n; ++i)
185  {
186  W[i]=0.0;
187  for (j=0; j < n ; ++j)
188  V[i][j]=0.0;
189  }
190 
191  // Householder reduction to bidiagonal form:
192  for ( i = 0 ; i < n ; ++i )
193  {
194  l = i + 1;
195  rv1[i] = scale * g;
196  g = s = scale = 0.0;
197  if ( i < m )
198  {
199  for ( k = i ; k < m ; ++k )
200  scale += fabs ( M[k][i] );
201  if ( scale )
202  {
203  for ( k = i ; k < m ; ++k )
204  {
205  M[k][i] /= scale;
206  s += M[k][i] * M[k][i];
207  }
208  f = M[i][i];
209  g = ( f >= 0.0 ) ? -fabs(sqrt(s)) : fabs(sqrt(s));
210  h = f * g - s;
211  M[i][i] = f - g;
212  for ( j = l ; j < n ; ++j )
213  {
214  for ( s = 0.0, k = i ; k < m ; ++k )
215  s += M[k][i] * M[k][j];
216  f = s / h;
217  for ( k = i ; k < m ; ++k )
218  M[k][j] += f * M[k][i];
219  }
220  for ( k = i ; k < m ; ++k )
221  M[k][i] *= scale;
222  }
223  }
224  W[i] = scale * g;
225  g = s = scale = 0.0;
226  if ( i < m && i != nm1 )
227  {
228  for ( k = l ; k < n ; ++k )
229  scale += fabs ( M[i][k] );
230  if ( scale )
231  {
232  for ( k = l ; k < n ; ++k )
233  {
234  M[i][k] /= scale;
235  s += M[i][k] * M[i][k];
236  }
237  f = M[i][l];
238  g = ( f >= 0.0 ) ? -fabs(sqrt(s)) : fabs(sqrt(s));
239  h = f * g - s;
240  M[i][l] = f - g;
241  for ( k = l ; k < n ; ++k )
242  rv1[k] = M[i][k] / h;
243  for ( j = l ; j < m ; ++j )
244  {
245  for ( s=0.0, k = l ; k < n ; ++k )
246  s += M[j][k] * M[i][k];
247  for ( k = l ; k < n ; ++k )
248  M[j][k] += s * rv1[k];
249  }
250  for ( k = l ; k < n ; ++k )
251  M[i][k] *= scale;
252  }
253  }
254  double tmp = fabs ( W[i] ) + fabs ( rv1[i] );
255  norm = ( norm > tmp ) ? norm : tmp;
256  }
257 
258  // accumulation of right-hand transformations:
259  for ( i = nm1 ; i >= 0 ; --i )
260  {
261  if ( i < nm1 )
262  {
263  if ( g )
264  {
265  for ( j = l ; j < n ; ++j )
266  V[j][i] = ( M[i][j] / M[i][l] ) / g;
267  for ( j = l ; j < n ; ++j )
268  {
269  for ( s = 0.0, k = l ; k < n ; ++k )
270  s += M[i][k] * V[k][j];
271  for ( k = l ; k < n ; ++k )
272  V[k][j] += s * V[k][i];
273  }
274  }
275  for ( j = l ; j < n ; ++j )
276  V[i][j] = V[j][i] = 0.0;
277  }
278  V[i][i] = 1.0;
279  g = rv1[i];
280  l = i;
281  }
282 
283  // accumulation of left-hand transformations:
284  for ( i = ( ( m < n ) ? m : n ) - 1 ; i >= 0 ; --i )
285  {
286  l = i + 1;
287  g = W[i];
288  for ( j = l ; j < n ; ++j )
289  M[i][j] = 0.0;
290  if ( g )
291  {
292  g = 1.0 / g;
293  for ( j = l ; j < n ; ++j )
294  {
295  for ( s = 0.0, k = l ; k < m ; ++k )
296  s += M[k][i] * M[k][j];
297  f = ( s / M[i][i] ) * g;
298  for ( k = i ; k < m ; ++k )
299  M[k][j] += f * M[k][i];
300  }
301  for ( j = i ; j < m ; ++j )
302  M[j][i] *= g;
303  }
304  else
305  for ( j = i ; j < m ; ++j )
306  M[j][i] = 0.0;
307  ++M[i][i];
308  }
309 
310  // diagonalization of the bidiagonal form:
311  for ( k = nm1 ; k >= 0 ; --k )
312  {
313  for ( its = 1 ; its <= NITER ; its++ )
314  {
315  flag = true;
316  for ( l = k ; l >= 0 ; l-- )
317  {
318  nm = l - 1;
319  if ( nm < 0 || fabs ( rv1[l]) + norm == norm )
320  {
321  flag = false;
322  break;
323  }
324  if ( fabs ( W[nm] ) + norm == norm )
325  break;
326  }
327  if ( flag )
328  {
329  c = 0.0;
330  s = 1.0;
331  for ( i = l ; i <= k ; i++ )
332  {
333  f = s * rv1[i];
334  rv1[i] = c * rv1[i];
335  if ( fabs(f) + norm == norm )
336  break;
337  g = W[i];
338 
339  absf = fabs(f);
340  absg = fabs(g);
341  h = ( absf > absg ) ?
342  absf * sqrt ( 1.0 + pow ( absg/absf, 2.0 ) ) :
343  ( ( absg==0 ) ? 0.0 : absg * sqrt ( 1.0 + pow ( absf/absg, 2.0 ) ) );
344 
345  W[i] = h;
346  h = 1.0 / h;
347  c = g * h;
348  s = -f * h;
349  for ( j = 0 ; j < m ; ++j )
350  {
351  y = M[j][nm];
352  z = M[j][ i];
353  M[j][nm] = y * c + z * s;
354  M[j][ i] = z * c - y * s;
355  }
356  }
357  }
358  z = W[k];
359  if ( l == k)
360  {
361  if ( z < 0.0 )
362  {
363  W[k] = -z;
364  for ( j = 0 ; j < n ; j++ )
365  V[j][k] = -V[j][k];
366  }
367  break; // this 'break' is always active if k==0
368  }
369  if ( its == NITER )
370  {
371  error_msg = "SVD_decomposition() error: no convergence in " +
372  NOMAD::itos ( NITER ) + " iterations";
373  delete [] rv1;
374  return false;
375  }
376  x = W[l];
377  nm = k - 1;
378  y = W[nm];
379  g = rv1[nm];
380  h = rv1[k];
381  f = ( (y-z) * (y+z) + (g-h) * (g+h) ) / ( 2.0 * h * y );
382 
383  absf = fabs(f);
384  g = ( absf > 1.0 ) ?
385  absf * sqrt ( 1.0 + pow ( 1.0/absf, 2.0 ) ) :
386  sqrt ( 1.0 + pow ( absf, 2.0 ) );
387 
388  f = ( (x-z) * (x+z) +
389  h * ( ( y / ( f + ( (f >= 0)? fabs(g) : -fabs(g) ) ) ) - h ) ) / x;
390  c = s = 1.0;
391 
392  for ( j = l ; j <= nm ; ++j )
393  {
394  i = j + 1;
395  g = rv1[i];
396  y = W[i];
397  h = s * g;
398  g = c * g;
399 
400  absf = fabs(f);
401  absh = fabs(h);
402  z = ( absf > absh ) ?
403  absf * sqrt ( 1.0 + pow ( absh/absf, 2.0 ) ) :
404  ( ( absh==0 ) ? 0.0 : absh * sqrt ( 1.0 + pow ( absf/absh, 2.0 ) ) );
405 
406  rv1[j] = z;
407  c = f / z;
408  s = h / z;
409  f = x * c + g * s;
410  g = g * c - x * s;
411  h = y * s;
412  y *= c;
413  for ( jj = 0 ; jj < n ; ++jj )
414  {
415  x = V[jj][j];
416  z = V[jj][i];
417  V[jj][j] = x * c + z * s;
418  V[jj][i] = z * c - x * s;
419  }
420 
421  absf = fabs(f);
422  absh = fabs(h);
423  z = ( absf > absh ) ?
424  absf * sqrt ( 1.0 + pow ( absh/absf, 2.0 ) ) :
425  ( ( absh==0 ) ? 0.0 : absh * sqrt ( 1.0 + pow ( absf/absh, 2.0 ) ) );
426 
427  W[j] = z;
428 
429  if ( z )
430  {
431  z = 1.0 / z;
432  c = f * z;
433  s = h * z;
434  }
435  f = c * g + s * y;
436  x = c * y - s * g;
437  for ( jj = 0 ; jj < m ; ++jj )
438  {
439  y = M[jj][j];
440  z = M[jj][i];
441  M[jj][j] = y * c + z * s;
442  M[jj][i] = z * c - y * s;
443  }
444  }
445  rv1[l] = 0.0;
446  rv1[k] = f;
447  W [k] = x;
448  }
449  }
450 
451  delete [] rv1;
452  return true;
453 }

456 /*--------------------------------------------------------------*/
457 /* LU decomposition */
458 /* Recoded from numerical recipes 3rd ed. 2007 */
459 /*--------------------------------------------------------------*/
460 /* */
461 /* M = L.U */
462 /* */
463 /* M ( n x n ) */
464 /* */
465 /* M is given as first argument and becomes LU */
466 /* LU = [[b00 b01 ... b0n] */
467 /* [a10 b11 ... b1n] */
468 /* [a20 a21 ... b2n] */
469 /* ........... */
470 /* [an0 an1 ... bnn] */
471 /* */
472 /*--------------------------------------------------------------*/
473 bool NOMAD::LU_decomposition ( std::string & error_msg,
474  double ** LU,
475  int n,
476  double & d,
477  int max_n ) // default=50
478 {
479  error_msg.clear();
480 
481  if ( max_n > 0 && n > max_n )
482  {
483  error_msg = "LU_decomposition() error: n > " + NOMAD::itos ( max_n );
484  return false;
485  }
486 
487  const double TINY = 1E-40;
488  int i, j, k;
489 
490  double big, temp;
491 
492  double * vv = new double[n]; // stores the implicit scaling of each row
493  int *indx = new int[n]; // stores the permutations
494 
495  d = 1; // No row interchange yet
496 
497  for (i = 0; i < n ; i++ ) // Loop over row to get implicit scaling information
498  {
499  big = 0.0;
500  for ( j = 0 ; j < n ; j++ )
501  {
502  if ( (temp = fabs(LU[i][j])) > big )
503  big = temp;
504  }
505  if ( big == 0 )
506  {
507  error_msg = "LU_decomposition() error: no nonzero largest element";
508  delete [] vv;
509  delete [] indx;
510 
511  return false;
512  }
513  vv[i]= 1.0/big; // Saves the scaling
514  }
515  for ( k = 0; k < n ; k++) // This is the outermost kij loop
516  {
517  big = 0.0; // Initialized the search for largest pivot element
518  int imax = k;
519  for ( i = k ; i < n ; i++ )
520  {
521  temp = vv[i]*fabs(LU[i][k]);
522  if ( temp > big )
523  {
524  big = temp;
525  imax = i;
526  }
527  }
528  if ( k != imax )
529  {
530  for ( j = 0; j < n ; j++ )
531  {
532  temp = LU[imax][j];
533  LU[imax][j]=LU[k][j];
534  LU[k][j]=temp;
535  }
536  d = -d;
537  vv[imax] = vv[k];
538  }
539  indx[k] = imax;
540  if ( LU[k][k] == 0.0) // TINY for zero
541  LU[k][k] = TINY;
542  for ( i = k+1 ; i < n ; i++ )
543  {
544  temp = LU[i][k] /= LU[k][k]; // Divide by pivot element
545  for ( j = k+1 ; j < n ; j++ ) // Innermos matrix: reduce remaining submatrix
546  LU[i][j] -= temp*LU[k][j];
547  }
548  }
549 
550  delete [] vv;
551  delete [] indx;
552  return true;
553 
554 }
