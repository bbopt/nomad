476 void NOMAD::Parameters::readParamFileAndSetEntries(const std::string &paramFile,
477  bool overwrite,
478  bool resetAllEntries )
479 {
480  // Warning:
481  // This method is static, so it cannot set _toBeChecked on the
482  // Parameters object. It will be set by readEntries().
483 
484  // Open the parameters file:
485  std::string err = "Could not open parameters file \'" + paramFile + "\'";
486  std::ifstream fin;
487  if ( access ( paramFile.c_str() , R_OK ) == 0 )
488  {
489  fin.open ( paramFile.c_str() );
490  if ( !fin.fail() )
491  {
492  err.clear();
493  }
494  }
495  if ( !err.empty() )
496  {
497  fin.close();
498  throw NOMAD::Exception(__FILE__, __LINE__, err);
499  }
500 
501  // the file is read: fill the set '_paramEntries' of ParameterEntry:
502  std::string line;
503  int line_number = 0;
504 
505  if ( resetAllEntries )
506  eraseAllEntries();
507 
508  while (fin.good() && !fin.eof())
509  {
510  line.clear();
511 
512  getline(fin, line);
513  line_number++;
514 
515  if (!fin.fail() && !line.empty())
516  {
517  readParamLine(line, paramFile, line_number, overwrite);
518  }
519  }
520 
521  // done with the file
522  fin.close();
523 }

553 void NOMAD::Parameters::readParamLine(const std::string &line,
554  const std::string &paramFile,
555  const int line_number,
556  bool overwrite)
557 {
558  auto pe = std::make_shared<NOMAD::ParameterEntry>(line);
559  if (nullptr == pe)
560  {
561  std::string err = "readParamLine: Error: Could not create parameter entry for parameter " + pe->getName();
562  throw NOMAD::Exception(paramFile, line_number, err);
563  }
564  pe->setParamFile(paramFile);
565  pe->setLine(line_number);
566 
567  if (pe->isOk())
568  {
569  if (overwrite)
570  {
571  std::shared_ptr<NOMAD::ParameterEntry> refPe = _paramEntries.find(pe->getName());
572  // Erase all parameters with this name.
573  if (nullptr != refPe)
574  {
575  _paramEntries.erase(refPe);
576  }
577  }
578  _paramEntries.insert(pe);
579  }
580  else
581  {
582  if (pe->getName() != "" && pe->getNbValues() == 0)
583  {
584  std::string err = "Invalid parameter: " + pe->getName();
585  // If reading a file (positive line number), throw an exception.
586  // Else, only show a warning.
587  if (line_number > 0)
588  {
589  throw NOMAD::Exception(paramFile, line_number, err);
590  }
591  else
592  {
593  std::cerr << "Warning: " << err << std::endl;
594  }
595  }
596  }
597 
598 }

602 /* read parameter entries */
603 /*----------------------------------------*/
604 void NOMAD::Parameters::readEntries(const bool overwrite)
605 {
606 
607  // parameters will have to be checked:
608  _toBeChecked = true;
609 
610  // interpret and set the entries using SET methods:
611  size_t sz;
612  int i;
613  NOMAD::Double d;
614  bool flag;
615  std::shared_ptr<NOMAD::ParameterEntry> pe;
616  std::string err;
617 
618  // First set DIMENSION from entries
619  std::string paramName = "DIMENSION";
620  auto attDim = getAttribute(paramName);
621 
622  if ( nullptr != attDim )
623  {
624  pe = _paramEntries.find(paramName);
625  if ( pe )
626  {
627  if ( !pe->isUnique() )
628  {
629  err = "Multiple entries of parameter: " + paramName + " at line #" + std::to_string(pe->getLine()) + "! ";
630  throw NOMAD::Exception(__FILE__,__LINE__, err);
631  }
632 
633  checkFormatSizeT(pe, sz);
634  setAttributeValue("DIMENSION", sz);
635  pe->setHasBeenInterpreted();
636  }
637  }
638  /*
639  else
640  {
641  err = "Missing mandatory DIMENSION parameter! ";
642  throw NOMAD::Exception(__FILE__,__LINE__, err);
643  }
644  */
645 
646  // Loop on all registered attributes.
647  // Set the attribute value from entries
648  // according to its type
649  for(const auto &att : getAttributes())
650  {
651  paramName = att->getName();
652  auto paramType = getAttributeType(paramName);
653  bool paramFromUniqueEntry = att->getParamFromUniqueEntry();
654  bool internal = att->isInternal();
655 
656  // Dimension already set
657  if (paramName == "DIMENSION")
658  {
659  continue;
660  }
661 
662  pe = _paramEntries.find(paramName);
663  while (pe)
664  {
665  // Test if internal
666  if (internal)
667  {
668  err = "Internal parameter: " + paramName + " at line #" + std::to_string(pe->getLine()) + " cannot be set in a parameter file! ";
669  throw NOMAD::Exception(__FILE__,__LINE__, err);
670  }
671 
672  // Test if multiple entries
673  if (paramFromUniqueEntry && !pe->isUnique() && !overwrite)
674  {
675  err = "Multiple entries of parameter: " + paramName + " at line #" + std::to_string(pe->getLine()) + "! ";
676  throw NOMAD::Exception(__FILE__,__LINE__, err);
677  }
678 
679  // BOOL
680  if (paramType == typeid(bool).name())
681  {
682  checkFormatBool(pe);
683  flag = NOMAD::stringToBool( *(pe->getValues().begin()) );
684  setAttributeValue(paramName, flag);
685  }
686  // SIZE_T
687  else if (paramType == typeid(size_t).name())
688  {
689  checkFormatSizeT(pe, sz);
690  setAttributeValue(paramName, sz);
691  }
692  // INTEGER
693  else if (paramType == typeid(int).name())
694  {
695  checkFormatInt(pe, i);
696  setAttributeValue(paramName, i);
697  }
698  // STRING
699  else if ( paramType == typeid(std::string).name() )
700  {
701  checkFormatString(pe);
702  setAttributeValue(paramName , *(pe->getValues().begin()) );
703  }
704  // ARRAY OF STRING
705  else if ( paramType == typeid(NOMAD::ArrayOfString).name() )
706  {
707  checkFormatArrayOfString(pe);
708  NOMAD::ArrayOfString aos(pe->getAllValues());
709  setAttributeValue(paramName, aos);
710  }
711  // BBInputTypeList
712  else if (paramType == typeid(NOMAD::BBInputTypeList).name())
713  {
714  checkFormat1(pe);
715  setAttributeValue(paramName, NOMAD::stringToBBInputTypeList(pe->getAllValues()));
716  }
717  // BBOutputTypeList
718  else if (paramType == typeid(NOMAD::BBOutputTypeList).name())
719  {
720  checkFormat1(pe);
721  setAttributeValue(paramName, NOMAD::stringToBBOutputTypeList(pe->getAllValues()));
722  }
723  // LHSearchType
724  else if (paramType == typeid(NOMAD::LHSearchType).name())
725  {
726  checkFormatNbEntries(pe, 2);
727  checkFormatAllSizeT(pe);
728  std::string allValues = pe->getAllValues();
729  auto lhSearch = NOMAD::LHSearchType(allValues);
730  setAttributeValue(paramName, lhSearch);
731  }
732  // SgtelibModelFeasibilityType
733  else if (paramType == typeid(NOMAD::SgtelibModelFeasibilityType).name())
734  {
735  checkFormat1(pe);
736  setAttributeValue(paramName, NOMAD::stringToSgtelibModelFeasibilityType(pe->getAllValues()));
737  }
738  // SgtelibModelFormulationType
739  else if (paramType == typeid(NOMAD::SgtelibModelFormulationType).name())
740  {
741  checkFormat1(pe);
742  setAttributeValue(paramName, NOMAD::stringToSgtelibModelFormulationType(pe->getAllValues()));
743  }
744  // EvalType
745  else if (paramType == typeid(NOMAD::EvalType).name())
746  {
747  checkFormat1(pe);
748  setAttributeValue(paramName, NOMAD::stringToEvalType(pe->getAllValues()));
749  }
750  // EvalSortType
751  else if (paramType == typeid(NOMAD::EvalSortType).name())
752  {
753  checkFormat1(pe);
754  setAttributeValue(paramName, NOMAD::stringToEvalSortType(pe->getAllValues()));
755  }
756  // DirectionType
757  else if (paramType == typeid(NOMAD::DirectionType).name())
758  {
759  checkFormat1(pe);
760  setAttributeValue(paramName, NOMAD::stringToDirectionType(pe->getAllValues()));
761  }
762  // DOUBLE
763  else if (paramType == typeid(NOMAD::Double).name())
764  {
765  checkFormatDouble(pe, d);
766  setAttributeValue(paramName , d);
767  }
768  // ARRAYOFDOUBLE; POINT
769  else if (paramType == typeid(NOMAD::ArrayOfDouble).name()
770  || paramType == typeid(NOMAD::Point).name())
771  {
772  if ( isAttributeDefaultValue<size_t>("DIMENSION") )
773  {
774  throw NOMAD::Exception(__FILE__,__LINE__,"Dimension must be set!");
775  }
776 
777  // Dimension is needed for ArrayOfDouble because
778  // several formats of the provided values are allowed
779  const size_t n = getAttributeValueProtected<size_t>("DIMENSION", false);
780 
781  NOMAD::ArrayOfDouble aod(n);
782  readValuesAsArray(*pe, aod);
783 
784  if ( paramType == typeid(NOMAD::Point).name() )
785  {
786  //Convert aod to Point before setting parameter
787  NOMAD::Point point(n);
788  for (size_t index = 0; index < n; index++)
789  {
790  point[index] = aod[index];
791  }
792  setAttributeValue(paramName, point);
793  }
794  else
795  {
796  setAttributeValue(paramName, aod);
797  }
798  }
799  // Vector of POINTS
800  else if (paramType == typeid(NOMAD::ArrayOfPoint).name())
801  {
802  if ( isAttributeDefaultValue<size_t>("DIMENSION") )
803  {
804  throw NOMAD::Exception(__FILE__,__LINE__,"Dimension must be set!");
805  }
806  const size_t n = getAttributeValueProtected<size_t>("DIMENSION", false);
807 
808  auto aopRef = getAttributeValueProtected<NOMAD::ArrayOfPoint>(paramName, false);
809  NOMAD::ArrayOfPoint aop = aopRef;
810 
811  NOMAD::Point updatePoint(n);
812  size_t pointIndex = readValuesForArrayOfPoint(*pe, updatePoint);
813  // If the aop at this index is already set, update it.
814  // Else, create it.
815  if (pointIndex < aop.size())
816  {
817  auto pointToUpdate = aop[pointIndex];
818  for (size_t index = 0; index < n; index++)
819  {
820  if (updatePoint[index].isDefined())
821  {
822  pointToUpdate[index] = updatePoint[index];
823  }
824  }
825  aop[pointIndex] = pointToUpdate;
826  }
827  else
828  {
829  aop.resize(pointIndex+1);
830  aop[pointIndex] = updatePoint;
831  }
832 
833  setAttributeValue(paramName, aop);
834  }
835  // List of VARIABLE_GROUP
836  else if (paramType == typeid(NOMAD::ListOfVariableGroup).name())
837  {
838  if ( isAttributeDefaultValue<size_t>("DIMENSION") )
839  {
840  throw NOMAD::Exception(__FILE__,__LINE__,"Dimension must be set!");
841  }
842  const size_t n = getAttributeValueProtected<size_t>("DIMENSION", false);
843 
844  auto lvg = getAttributeValueProtected<NOMAD::ListOfVariableGroup>(paramName, false);
845 
846  NOMAD::VariableGroup aVariableGroup;
847  size_t nbIndex = readValuesForVariableGroup(*pe, aVariableGroup);
848  // If the aop at this index is already set, update it.
849  // Else, create it.
850  if (nbIndex < n)
851  {
852  lvg.push_back(aVariableGroup);
853  }
854  else
855  {
856  err = "Number of indices for VARIABLE_GROUP must be smaller than DIMENSION (" + std::to_string(n) + ").";
857  throw NOMAD::Exception(__FILE__,__LINE__, err);
858  }
859 
860  setAttributeValue(paramName, lvg);
861  }
862  else
863  {
864  err = "Parameter " + paramName + " has been registered but its type cannot be read";
865  throw NOMAD::Exception(__FILE__,__LINE__, err);
866  }
867  pe->setHasBeenInterpreted();
868  // Get next parameter entry with this name, if multiple entries are permitted
869  pe = pe->getNext();
870  }
871  }

1039 size_t NOMAD::Parameters::readValuesForVariableGroup(const NOMAD::ParameterEntry &pe,
1040  NOMAD::VariableGroup &vg )
1041 {

1043 
1044  std::list<std::string>::const_iterator it , end;
1045  std::pair<NOMAD::VariableGroup::iterator,bool> ret;
1046  // just one variable index (can be '*' or a range of indices 'i-j'):
1047  if ( pe.getNbValues() == 1 )

1049  size_t j, k;
1050  it = pe.getValues().begin();

1071  else
1072  {
1073  end = pe.getValues().end();

1078  std::string err = "Invalid format for index list: ";
1079  err += pe.getName() + " at line " + std::to_string(pe.getLine());

1090  }
1091

1239  }

1266  throw NOMAD::Exception(__FILE__,__LINE__, err);
