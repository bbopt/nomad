66 void NOMAD::Algorithm::init()
67 {
68  _name = "AGenericAlgorithmHasNoName";
69 
70  // Verifications that throw Exceptions to the Constructor if not validated.
71  verifyParentNotNull();
72 
73  if (nullptr == _runParams)
74  {
75  throw NOMAD::StepException(__FILE__, __LINE__,
76  "A valid RunParameters must be provided to an Algorithm constructor.", this);
77  }
78 
79  if (nullptr == _pbParams)
80  {
81  throw NOMAD::StepException(__FILE__, __LINE__,
82  "A valid PbParameters must be provided to the Algorithm constructor.", this);
83  }
84 
85  if ( nullptr == _stopReasons )
86  throw NOMAD::StepException(__FILE__, __LINE__,
87  "Valid stop reasons must be provided to the Algorithm constructor.", this);
88 
89 #ifdef _OPENMP
90  omp_init_lock(&_algoCommentLock);
91 #endif // _OPENMP
92 
93  // Check pbParams if needed, ex. if a copy of PbParameters was given to the Algorithm constructor.
94  _pbParams->checkAndComply();
95 
96  // Instantiate generic algorithm termination
97  _termination = std::make_unique<NOMAD::Termination>( this );
98 
99  // Update SubproblemManager
100  NOMAD::Point fullFixedVariable = isRootAlgo() ? _pbParams->getAttributeValue<NOMAD::Point>("FIXED_VARIABLE")
101  : NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(_parentStep);
102 
103  NOMAD::Subproblem subproblem(_pbParams, fullFixedVariable);
104  NOMAD::SubproblemManager::getInstance()->addSubproblem(this, subproblem);
105  _pbParams = subproblem.getPbParams();
106  _pbParams->checkAndComply();
107 
108  /** Step::userInterrupt() will be called if CTRL-C is pressed.
109  * Currently, the main thread will wait for all evaluations to be complete.
110  * \todo Propage interruption to all threads, for all parallel evaluations of blackbox.
111  */
112  signal(SIGINT, userInterrupt);
113  signal(SIGSEGV, debugSegFault);
114 
115 }

127 // Set _algoComment, and push the current algo comment to _prevAlgoComment pile.
128 // Do not push an empty string on an empty pile, it is irrelevant.
129 // If force = true, do not set a new algo comment until this comment is reset.
130 void NOMAD::Algorithm::setAlgoComment(const std::string& algoComment, const bool force)
131 {
132  if (isSubAlgo())
133  {
134  auto rootAlgo = const_cast<NOMAD::Algorithm*>(getRootAlgorithm());
135  rootAlgo->setAlgoComment(algoComment, force);
136  }
137  else
138  {
139 #ifdef _OPENMP
140  omp_set_lock(&_algoCommentLock);
141 #endif // _OPENMP
142  if (!_forceAlgoComment)
143  {
144  // Push algo comment to _prevAlgoComment
145  if (!_prevAlgoComment.empty() || !_algoComment.empty())
146  {
147  _prevAlgoComment.push_back(_algoComment);
148  }
149  _algoComment = algoComment;
150  }
151  if (force)
152  {
153  _forceAlgoComment = true;
154  }
155 #ifdef _OPENMP
156  omp_unset_lock(&_algoCommentLock);
157 #endif // _OPENMP
158  }
159 }

162 // Pop the previous algo comment from the _prevAlgoComment pile.
163 void NOMAD::Algorithm::resetPreviousAlgoComment(const bool force)
164 {
165  if (isSubAlgo())
166  {
167  auto rootAlgo = const_cast<NOMAD::Algorithm*>(getRootAlgorithm());
168  rootAlgo->resetPreviousAlgoComment(force);
169  }
170  else
171  {
172 #ifdef _OPENMP
173  omp_set_lock(&_algoCommentLock);
174 #endif // _OPENMP
175  if (!_forceAlgoComment || force)
176  {
177  if (_prevAlgoComment.empty())
178  {
179  _algoComment = "";
180  }
181  else
182  {
183  // Remove last element, simulate a "pop".
184  _algoComment = std::move(_prevAlgoComment[_prevAlgoComment.size()-1]);
185  _prevAlgoComment.erase(_prevAlgoComment.end()-1);
186  }
187  if (_forceAlgoComment)
188  {
189  _forceAlgoComment = false;
190  }
191  }
192 #ifdef _OPENMP
193  omp_unset_lock(&_algoCommentLock);
194 #endif // _OPENMP
195  }
196 }

215 void NOMAD::Algorithm::startImp()
216 {
217 #ifdef TIME_STATS
218  if (isRootAlgo())
219  {
220  _startTime = NOMAD::Clock::getCPUTime();
221  }
222 #endif // TIME_STATS
223  // Comment to appear at the end of stats lines
224  // By default, nothing is added
225  setAlgoComment("");
226 
227  // All stop reasons are reset.
228  _stopReasons->setStarted();
229 
230  // SuccessType is reset
231  _algoSuccessful = false;
232  _algoBestSuccess = NOMAD::SuccessType::NOT_EVALUATED;
233 
234  if (isRootAlgo())
235  {
236  // Update hot restart info
237  readInformationForHotRestart();
238  NOMAD::CacheBase::getInstance()->setStopWaiting(false);
239  }
240 
241  // By default reset the lap counter for BbEval and set the lap maxBbEval to INF
242  NOMAD::EvcInterface::getEvaluatorControl()->resetLapBbEval();
243  NOMAD::EvcInterface::getEvaluatorControl()->setLapMaxBbEval( NOMAD::INF_SIZE_T );
244  NOMAD::EvcInterface::getEvaluatorControl()->resetModelEval();
245 
246  if (nullptr == _megaIteration)
247  {
248  // Default behavior - not hot restart.
249  // Clear cache hits.
250  // Initialization.
251  // Eval X0s.
252 
253  if (isRootAlgo())
254  {
255  // Ensure we do not count cache hits which may have been read in the cache.
256  NOMAD::CacheBase::resetNbCacheHits();
257  }
258 
259  // Perform algo initialization only when available.
260  if (nullptr != _initialization)
261  {
262  _initialization->start();
263  _initialization->run();
264  _initialization->end();
265  }
266 
267  }
268  else
269  {
270  // Hot restart situation.
271  // We will not need Initialization.
272  auto barrier = _megaIteration->getBarrier();
273 
274  // Update X0s
275  // Use best points.
276  auto bestPoints = barrier->getAllPoints();
277 
278  NOMAD::ArrayOfPoint x0s;
279  if (!bestPoints.empty())
280  {
281  std::transform(bestPoints.begin(), bestPoints.end(), std::back_inserter(x0s),
282  [](NOMAD::EvalPoint evalPoint) -> NOMAD::EvalPoint { return evalPoint; });
283 
284  }
285  _pbParams->setAttributeValue<NOMAD::ArrayOfPoint>("X0", x0s);
286  _pbParams->checkAndComply();
287  }
288 }

291 void NOMAD::Algorithm::endImp()
292 {
293  if ( _endDisplay )
294  {
295  displayBestSolutions();
296 #ifdef TIME_STATS
297  if (isRootAlgo())
298  {
299  _totalRealAlgoTime = NOMAD::Clock::getTimeSinceStart();
300  _totalCPUAlgoTime += NOMAD::Clock::getCPUTime() - _startTime;
301  }
302 #endif // TIME_STATS
303 
304  displayEvalCounts();
305  }
306 
307  // Update the SearchMethod success type with best success found.
308  if ( _algoSuccessful )
309  {
310  // The parent can be a SearchMethod (NM-Mads Search) or not (that is NM is a standalone optimization)
311  auto searchMethodConst = dynamic_cast<const NOMAD::SearchMethodBase*>(_parentStep);
312 
313  if (searchMethodConst != nullptr)
314  {
315  auto searchMethod = const_cast<NOMAD::SearchMethodBase*>(searchMethodConst);
316  searchMethod->setSuccessType(_algoBestSuccess);
317  }
318 
319  }
320 
321  // By default reset the lap counter for BbEval and set the lap maxBbEval to INF
322  NOMAD::EvcInterface::getEvaluatorControl()->resetLapBbEval();
323  NOMAD::EvcInterface::getEvaluatorControl()->setLapMaxBbEval( NOMAD::INF_SIZE_T );
324 
325  if (isRootAlgo())
326  {
327  saveInformationForHotRestart();
328  NOMAD::CacheBase::getInstance()->setStopWaiting(true);
329  }
330 
331  // Reset stats comment
332  resetPreviousAlgoComment();
333 }

356 void NOMAD::Algorithm::saveInformationForHotRestart() const
357 {
358  // If we want to stop completely and then be able to restart
359  // from where we were, we need to save some information on file.
360  //
361  // Issue 372: Maybe we need to write current parameters. If we write them,
362  // ignore initial values, only take latest values down the Parameter tree.
363  // For now, using initial parameters.
364 
365  // Cache file is treated independently from hot restart file.
366  // As long as the cache file name is set, it is written.
367  // This is the behavior of NOMAD 3.
368  std::string cacheFile = NOMAD::CacheBase::getInstance()->getFileName();
369  if (!cacheFile.empty())
370  {
371  NOMAD::CacheBase::getInstance()->write();
372  }
373  if ( _runParams->getAttributeValue<bool>("HOT_RESTART_WRITE_FILES"))
374  {
375  std::cout << "Save information for hot restart." << std::endl;
376  std::cout << "Write hot restart file." << std::endl;
377  NOMAD::write(*this, _runParams->getAttributeValue<std::string>("HOT_RESTART_FILE"));
378  }

382 void NOMAD::Algorithm::displayBestSolutions() const
383 {
384  std::vector<NOMAD::EvalPoint> evalPointList;
385  // Display best feasible solutions.
386  std::string sFeas;
387  // Output level is very high if there are no parent algorithm
388  // Output level is info if this algorithm is a sub part of another algorithm.
389  NOMAD::OutputLevel outputLevel = isSubAlgo() ? NOMAD::OutputLevel::LEVEL_INFO
390  : NOMAD::OutputLevel::LEVEL_VERY_HIGH;
391  auto solFormat = NOMAD::OutputQueue::getInstance()->getSolFormat();
392  auto computeType = NOMAD::EvcInterface::getEvaluatorControl()->getComputeType();
393  if (isRootAlgo())
394  {
395  solFormat.set(-1);
396  }
397  NOMAD::OutputInfo displaySolFeas(_name, sFeas, outputLevel);
398  auto fixedVariable = NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this);
399 
400  sFeas = "Best feasible solution";
401  auto barrier = getMegaIterationBarrier();
402  if (nullptr != barrier)
403  {
404  evalPointList = barrier->getAllXFeas();
405  NOMAD::convertPointListToFull(evalPointList, fixedVariable);
406  }
407  size_t nbBestFeas = evalPointList.size();
408 
409  if (0 == nbBestFeas)
410  {
411  sFeas += ": Undefined.";
412  displaySolFeas.addMsg(sFeas);
413  }
414  else if (1 == nbBestFeas)
415  {
416  sFeas += ": ";
417  displaySolFeas.addMsg(sFeas + evalPointList[0].display(computeType, solFormat, NOMAD::DISPLAY_PRECISION_FULL));
418  }
419  else
420  {
421  sFeas += "s: ";
422  displaySolFeas.addMsg(sFeas + evalPointList[0].display(computeType, solFormat, NOMAD::DISPLAY_PRECISION_FULL));
423  }
424 
425 
426  const size_t maxSolCount = 8;
427  if (nbBestFeas > 1)
428  {
429  std::vector<NOMAD::EvalPoint>::const_iterator it;
430  size_t solCount = 0;
431  for (it = evalPointList.begin(); it != evalPointList.end(); ++it)
432  {
433  solCount++;
434  if (evalPointList.begin() == it)
435  {
436  continue; // First element already added
437  }
438  sFeas = " ";
439  displaySolFeas.addMsg(sFeas + it->display(computeType, solFormat, NOMAD::DISPLAY_PRECISION_FULL));
440  if (solCount >= maxSolCount)
441  {
442  // We printed enough solutions already.
443  displaySolFeas.addMsg("... A total of " + std::to_string(evalPointList.size()) + " feasible solutions were found.");
444  break;
445  }
446  }
447  }
448 
449  NOMAD::OutputQueue::Add(std::move(displaySolFeas));
450 
451  evalPointList.clear();
452 
453 
454  // Display best infeasible solutions.
455  std::string sInf;
456  NOMAD::OutputInfo displaySolInf(_name, sInf, outputLevel);
457  sInf = "Best infeasible solution";
458  if (nullptr != barrier)
459  {
460  evalPointList = barrier->getAllXInf();
461  NOMAD::convertPointListToFull(evalPointList, fixedVariable);
462  }
463  size_t nbBestInf = evalPointList.size();
464 
465  if (0 == nbBestInf)
466  {
467  sInf += ": Undefined.";
468  displaySolInf.addMsg(sInf);
469  }
470  else if (1 == nbBestInf)
471  {
472  sInf += ": ";
473  displaySolInf.addMsg(sInf + evalPointList[0].display(computeType, solFormat, NOMAD::DISPLAY_PRECISION_FULL));
474  }
475  else
476  {
477  sInf += "s: ";
478  displaySolInf.addMsg(sInf + evalPointList[0].display(computeType, solFormat, NOMAD::DISPLAY_PRECISION_FULL));
479  }
480 
481  if (nbBestInf > 1)
482  {
483  size_t solCount = 0;
484  std::vector<NOMAD::EvalPoint>::const_iterator it;
485  for (it = evalPointList.begin(); it != evalPointList.end(); ++it)
486  {
487  solCount++;
488  if (evalPointList.begin() == it)
489  {
490  continue; // First element already added
491  }
492  displaySolInf.addMsg(" " + it->display(computeType, solFormat, NOMAD::DISPLAY_PRECISION_FULL));
493  if (solCount >= maxSolCount)
494  {
495  // We printed enough solutions already.
496  displaySolInf.addMsg("... A total of " + std::to_string(evalPointList.size()) + " infeasible solutions were found.");
497  break;
498  }
499  }
500  }
501 
502  NOMAD::OutputQueue::Add(std::move(displaySolInf));
503 }

506 void NOMAD::Algorithm::displayEvalCounts() const
507 {
508  // Display evaluation information
509 
510  // Used to display or not certain values
511  bool isSub = isSubAlgo();
512 
513  // Actual numbers
514  size_t bbEval = NOMAD::EvcInterface::getEvaluatorControl()->getBbEval();
515  size_t lapBbEval = NOMAD::EvcInterface::getEvaluatorControl()->getLapBbEval();
516  size_t nbEval = NOMAD::EvcInterface::getEvaluatorControl()->getNbEval();
517  size_t modelEval = NOMAD::EvcInterface::getEvaluatorControl()->getModelEval();
518  size_t totalModelEval = NOMAD::EvcInterface::getEvaluatorControl()->getTotalModelEval();
519  size_t nbCacheHits = NOMAD::CacheBase::getNbCacheHits();
520  int nbEvalNoCount = static_cast<int>(nbEval - bbEval - nbCacheHits);
521 
522  // What needs to be shown, according to the counts and to the value of isSub
523  bool showNbEvalNoCount = (nbEvalNoCount > 0);
524  bool showModelEval = isSub && (modelEval > 0);
525  bool showTotalModelEval = (totalModelEval > 0);
526  bool showNbCacheHits = (nbCacheHits > 0);
527  bool showNbEval = (nbEval > bbEval);
528  bool showLapBbEval = isSub && (bbEval > lapBbEval && lapBbEval > 0);

531  NOMAD::OutputLevel outputLevelHigh = isSub ? NOMAD::OutputLevel::LEVEL_INFO
532  : NOMAD::OutputLevel::LEVEL_HIGH;
533  NOMAD::OutputLevel outputLevelNormal = isSub ? NOMAD::OutputLevel::LEVEL_INFO
534  : NOMAD::OutputLevel::LEVEL_NORMAL;

536  // Padding for nice presentation

538  sFeedTotalModelEval, sFeedCacheHits, sFeedNbEval;
539

559  sFeedNbEval += " ";

569  //sFeedNbEval += "";
570  }
571  else if (showTotalModelEval)
572  {
573  sFeedBbEval += " ";
574  //sFeedLapBbEval += "";
575  //sFeedNbEvalNoCount += "";
576  //sFeedModelEval += " ";

579  //sFeedNbEval += "";

581

603  }
604  if (showNbEvalNoCount)

606  AddOutputInfo(sNbEvalNoCount, outputLevelNormal);

608  if (showModelEval)

610  AddOutputInfo(sModelEval, outputLevelNormal);
611  }

615  }

622  AddOutputInfo(sNbEval, outputLevelNormal);
623  }

631 }
