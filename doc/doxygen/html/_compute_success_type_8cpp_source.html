58 NOMAD::SuccessType NOMAD::ComputeSuccessType::defaultComputeSuccessType(
59  const std::shared_ptr<NOMAD::EvalPoint>& evalPoint1,
60  const std::shared_ptr<NOMAD::EvalPoint>& evalPoint2,
61  const NOMAD::Double& hMax)
62 {
63  NOMAD::SuccessType success = NOMAD::SuccessType::NOT_EVALUATED;
64 
65  if (nullptr != evalPoint1)
66  {
67  if (nullptr == evalPoint2)
68  {
69  if (evalPoint1->getH(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD) > hMax)
70  {
71  // Even if evalPoint2 is NULL, this case is still
72  // not a success.
73  success = NOMAD::SuccessType::UNSUCCESSFUL;
74  }
75  else if (evalPoint1->isFeasible(NOMAD::EvalType::BB))
76  {
77  // New feasible point: full success
78  success = NOMAD::SuccessType::FULL_SUCCESS;
79  }
80  else
81  {
82  // New infeasible makes for partial success, not full success
83  success = NOMAD::SuccessType::PARTIAL_SUCCESS;
84  }
85  }
86  else
87  {
88  success = NOMAD::Eval::computeSuccessType(evalPoint1->getEval(NOMAD::EvalType::BB),
89  evalPoint2->getEval(NOMAD::EvalType::BB),
90  NOMAD::ComputeType::STANDARD,
91  hMax);
92  }
93  }
94 
95  return success;
96 }

125 NOMAD::SuccessType NOMAD::ComputeSuccessType::computeSuccessTypeModel(
126  const std::shared_ptr<NOMAD::EvalPoint>& evalPoint1,
127  const std::shared_ptr<NOMAD::EvalPoint>& evalPoint2,
128  const NOMAD::Double& hMax)
129 {
130  NOMAD::SuccessType success = NOMAD::SuccessType::NOT_EVALUATED;
131  const NOMAD::EvalType evalTypeModel = NOMAD::EvalType::MODEL;
132 
133  if (nullptr != evalPoint1)
134  {
135  if (evalPoint1->getH(evalTypeModel, NOMAD::ComputeType::STANDARD) > hMax)
136  {
137  success = NOMAD::SuccessType::UNSUCCESSFUL;
138  }
139  else if (nullptr == evalPoint2)
140  {
141  success = NOMAD::SuccessType::FULL_SUCCESS;
142  }
143  else
144  {
145  success = NOMAD::Eval::computeSuccessType(evalPoint1->getEval(evalTypeModel),
146  evalPoint2->getEval(evalTypeModel),
147  NOMAD::ComputeType::STANDARD,
148  hMax);
149  }
150  }
151 
152  return success;
153 }
