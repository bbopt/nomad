113 // Set of indexes for neighbour creation
114 void NOMAD::Projection::buildIndexSet(const size_t n)
115 {
116  const size_t nbNeighbours = static_cast<size_t>(pow(2.0, n));
117  // Build the set of indexes
118  if (_nbProjTrial < nbNeighbours * 1.3)
119  {
120  // Select randomly _nbProjTrial indexes
121  for (size_t i = 0; i < _nbProjTrial; i++)
122  {
123  _indexSet.insert(NOMAD::RNG::rand() % nbNeighbours);
124  }
125  }
126  else
127  {
128  // Use all indexes
129  for (size_t i = 0 ; i < nbNeighbours; i++)
130  {
131  _indexSet.insert(i);
132  }
133  }

165 void NOMAD::Projection::projectPoint(const NOMAD::EvalPoint& oraclePoint)
166 {
167  std::string s;
168 
169  // Non projected point
170  nonProjectedPoint(oraclePoint);
171 
172  // STD projected point
173  stdProjectedPoint(oraclePoint);
174  // TODO compute this otherwise, if needed
175  //auto f = bestEvalPoint.getF(evalType);
176  //auto h = bestEvalPoint.getH(evalType);
177 
178  // Try pertubation
179  std::string subStepName = "Projection candidate";
180  AddOutputInfo(subStepName, true, false);
181 
182  NOMAD::EvalPointSet trySet;
183 
184  NOMAD::CacheInterface cacheInterface(this);
185  std::vector<NOMAD::EvalPoint> evalPointList;
186  size_t nbCachePoints = cacheInterface.getAllPoints(evalPointList);
187 
188  for (auto index : _indexSet)
189  {
190  // Compute perturbation
191  auto perturbation = computePerturbation(oraclePoint, index);
192 
193  // Loop on points of the cache
194  for (auto xRef : evalPointList)
195  {
196  auto xTry = buildProjectionTrialPoint(xRef, perturbation);
197  trySet.insert(xTry);
198  }
199  }
200 
201  s = std::to_string(_indexSet.size()) + " perturbation vectors";
202  NOMAD::OutputQueue::Add(s, _displayLevel);
203  s = std::to_string(nbCachePoints) + " cache points";
204  NOMAD::OutputQueue::Add(s, _displayLevel);
205  s = std::to_string(trySet.size()) + " projection candidates";
206  NOMAD::OutputQueue::Add(s, _displayLevel);
207 
208  AddOutputInfo(subStepName, false, true);
209 
210  // Greedy selection
211  const size_t p = trySet.size();
212  std::vector<bool> keep;
213  keep.resize(p);
214 
215  if (p <= _nbProjTrial)
216  {
217  // Greedy selection not needed - keep all
218  for (size_t i = 0; i < p; i++)
219  {
220  keep[i] = true;
221  }
222  }
223  else
224  {
225  // DO the greedy selection
226  doGreedySelection(oraclePoint, trySet, keep);
227  }
228 
229  for (auto xTry : trySet)
230  {
231  insertTrialPoint(xTry);
232  }
233 
234  _indexSet.clear();
235  trySet.clear();
236  keep.clear();
237

240  // in the dynamic (quad or sgtelib) model
241  // TODO Analyse from NOMAD 3 and see if we can do something similiar
242  // in NOMAD 4. It may not be worth it, it seems more like an
243  // issue of sorting the points accorting to a SgtelibModel, and
244  // that would be better done in the EvaluatorControl.
245  //evaluateProjectionTrialPoints(trySet, ev, keep, bestEvalPoint);

298 // Compute perturbation
299 NOMAD::Direction NOMAD::Projection::computePerturbation(const NOMAD::EvalPoint& oraclePoint,
300  size_t index)
301 {
302  NOMAD::Direction perturbationDir(oraclePoint.size());
303 
304  for (size_t j = 0; j < oraclePoint.size(); j++)
305  {
306  NOMAD::Double dmj = _mesh->getdeltaMeshSize(j); // Get the value of delta_m
307 
308  // Inverse dmi depending on parity of index
309  if (index & 1)
310  {
311  dmj *= -1;
312  }
313 
314  // Set perturbation
315  perturbationDir[j] = dmj;
316 
317  // Right shift ie: divide by 2
318  index = (index >> 1);
319 
320  } // End of the construction of the perturbation
321 
322  return perturbationDir;
323 }

342 void NOMAD::Projection::doGreedySelection(const NOMAD::EvalPoint &oraclePoint,
343  const NOMAD::EvalPointSet& trySet,
344  std::vector<bool>& keep)
345 {
346  const size_t p = trySet.size();
347  const size_t n = oraclePoint.size();
348  std::string s;
349 
350  int inew;
351  double * xnew_d;
352  double * xref_d = new double [n];
353  for (size_t j = 0 ; j < n ; j++)
354  {
355  xref_d[j] = oraclePoint[j].todouble();
356  }
357 
358  // Convert set of trial candidates into an array
359  //--------------------------------
360  double ** X = new double * [p];
361  size_t i = 0;
362  for (auto xTry : trySet)
363  {
364  X[i] = new double [n];
365  for (size_t j = 0; j < n; j++)
366  {
367  X[i][j] = xTry[j].todouble();
368  }
369 
370  i++;
371  }
372 
373  // Distance to xref
374  double * Dr = new double [p];
375  // Distance to the set of selected (or kept) points
376  double * Ds = new double [p];
377  double * Ds2 = new double [p]; // Square of Ds
378 
379  // First selected point
380  inew = 0;
381  xnew_d = X[inew];
382  double lambda = 3;
383 
384  // Buffer for dxj
385  double dxj;
386 
387 
388  // Initialise the distances
389  //--------------------------------------
390  for (i = 0 ; i < p ; i++)
391  {
392  // Compute distance between each point of the set and the point xref
393  // and also between each point of the set and the selected point xnew
394  Dr[i] = 0;
395  Ds[i] = 0;
396  for (size_t j = 0 ; j < n ; j++ )
397  {
398  double dmj = _mesh->getdeltaMeshSize(j).todouble();
399  dxj = (xref_d[j]-X[i][j])/dmj;
400  Dr[i] += dxj*dxj;
401  dxj = (xnew_d[j]-X[i][j])/dmj;
402  Ds[i] += dxj*dxj;
403  }
404  Dr[i] = sqrt(Dr[i]);
405  Ds2[i] = Ds[i];
406  Ds[i] = sqrt(Ds[i]);
407  // Init keep
408  keep[i] = false;
409  }
410 
411  // Note that we selected the first point
412  size_t nbKeep = 1;
413  keep[inew] = true;
414 
415  // Greedy selection
416  //---------------------------
417  while (nbKeep < _nbProjTrial)
418  {
419  // Find the index that maximizes Ds-lambda*Dr
420  double Q_max = -INF;
421  inew = 1;
422  for (i = 0; i < p; i++)
423  {
424  // Q is the selection criteria: Q = Ds-lambda*Dr
425  double Q = Ds[i]-lambda*Dr[i];
426  if ( Q > Q_max )
427  {
428  inew = i;
429  Q_max = Q;
430  }
431  }
432 
433  if ( Ds[inew] == 0 )
434  {
435  // If the point is already in the set, then reduce lambda
436  lambda *= 0.9;
437  if (lambda < 1e-6)
438  {
439  break;
440  }
441  }
442  else
443  {
444  // Otherwise, add the point to the set
445  keep[inew] = true;
446  nbKeep++;
447  xnew_d = X[inew];
448  // Update its distance to the set
449  Ds[inew] = 0;
450  // Update the other distances to the set
451  for (i = 0; i < p; i++)
452  {
453  if (Ds[i] > 0)
454  {
455  // Compute distance between each point of the set and the point xref
456  // and also between each point of the set and the selected point xnew
457  double d = 0;
458  for (size_t j = 0; j < n; j++)
459  {
460  double dmj = _mesh->getdeltaMeshSize(j).todouble();
461  dxj = (xnew_d[j]-X[i][j])/dmj;
462  d += dxj * dxj;
463  }
464  if (Ds2[i] > d)
465  {
466  Ds2[i] = d;
467  Ds[i] = sqrt(d);
468  }
469  }
470  }
471  }
472  } // End while
473 
474 
475  s = std::to_string(nbKeep) + " projection candidates (after greedy selection)";
476  NOMAD::OutputQueue::Add(s, _displayLevel);
477 
478  delete [] xref_d;
479  delete [] Ds;
480  delete [] Ds2;
481  delete [] Dr;
482  for (i = 0; i < p; i++)
483  {
484  delete [] X[i];
485  }
486  delete [] X;
487 
488 }
489 // End doGreedySelection
