62 void NOMAD::SgtelibModel::init()
63 {
64  setName("SgtelibModel");
65  verifyParentNotNull();
66 
67  auto modelFormulation = _runParams->getAttributeValue<NOMAD::SgtelibModelFormulationType>("SGTELIB_MODEL_FORMULATION");
68  auto modelFeasibility = _runParams->getAttributeValue<NOMAD::SgtelibModelFeasibilityType>("SGTELIB_MODEL_FEASIBILITY");
69  auto modelDefinition = _runParams->getAttributeValue<NOMAD::ArrayOfString>("SGTELIB_MODEL_DEFINITION");
70 
71  if (NOMAD::SgtelibModelFormulationType::EXTERN == modelFormulation)
72  {
73  // Extern model. Early out.
74  return;
75  }
76 
77  // Check
78  if ( (NOMAD::SgtelibModelFormulationType::FS == modelFormulation)
79  || (NOMAD::SgtelibModelFormulationType::EIS == modelFormulation) )
80  {
81  if (NOMAD::SgtelibModelFeasibilityType::C != modelFeasibility)
82  {
83  std::cerr << "ERROR : Formulations FS and EIS can only be used with FeasibilityMethod C" << std::endl;
84  throw SGTELIB::Exception(__FILE__, __LINE__, "SgtelibModel: SGTELIB_MODEL_FEASIBILITY not valid");
85  }
86  }
87 
88  // Count the number of constraints
89  const auto bbot = NOMAD::SgtelibModel::getBBOutputType();
90  size_t nbConstraints = NOMAD::getNbConstraints(bbot);
91  _nbModels = getNbModels(modelFeasibility, nbConstraints);
92 
93 
94  // Init the TrainingSet
95  size_t n = _pbParams->getAttributeValue<size_t>("DIMENSION");
96  SGTELIB::Matrix empty_X("empty_X", 0, static_cast<int>(n));
97  SGTELIB::Matrix empty_Z("empty_Z", 0, static_cast<int>(_nbModels));
98  _trainingSet = std::make_shared<SGTELIB::TrainingSet>(empty_X, empty_Z);
99 
100  // Build the Sgtelib Model
101  _model = std::shared_ptr<SGTELIB::Surrogate>(SGTELIB::Surrogate_Factory(*_trainingSet, modelDefinition.display()));
102 
103  // Instantiate sgtelib model initialization class
104  _initialization = std::make_unique<NOMAD::SgtelibModelInitialization>(this);
105 }

136 /*-------------------------------------------------------------*/
137 /* isReady */
138 /* Return a boolean to know if the SgtelibModel model has been */
139 /* build and can be called for predictions. */
140 /*-------------------------------------------------------------*/
141 bool NOMAD::SgtelibModel::isReady() const
142 {
143  bool retReady = _ready;
144 
145  if (!retReady)
146  {
147  auto modelFormulation = _runParams->getAttributeValue<NOMAD::SgtelibModelFormulationType>("SGTELIB_MODEL_FORMULATION");
148  if (NOMAD::SgtelibModelFormulationType::EXTERN == modelFormulation)
149  {
150  // Extern model.
151  _ready = true;
152  retReady = true;
153  }
154  }
155 
156  if (!retReady)
157  {
158  if (!_trainingSet)
159  {
160  throw NOMAD::Exception ( __FILE__, __LINE__ ,
161  "SgtelibModel::isReady : no training set");
162  }
163 
164  if ( _trainingSet->is_ready())
165  {
166  const int pvar = _trainingSet->get_pvar();
167  _ready = ( _model->is_ready() ) && ( pvar > 10 );
168  retReady = _ready;
169  }
170  }
171 
172  return retReady;
173 }

212 // Used during SgtelibModelUpdate step.
213 // Update the bounds of the model.
214 void NOMAD::SgtelibModel::setModelBounds(std::shared_ptr<SGTELIB::Matrix> X)
215 {
216  size_t n = _pbParams->getAttributeValue<size_t>("DIMENSION");
217  if (n != (size_t)X->get_nb_cols())
218  {
219  throw NOMAD::Exception(__FILE__, __LINE__,
220  "SgtelibModel::setModelBounds() dimensions do not match");
221  }
222 
223  int nbDim = X->get_nb_cols();
224  int nbPoints = X->get_nb_rows();
225 
226  // Build model bounds
227  NOMAD::Double lb;
228  NOMAD::Double ub;
229 
230  for (int j = 0; j < nbDim; j++)
231  {
232  lb = _modelLowerBound[j];
233  ub = _modelUpperBound[j];
234  for (int p = 0; p < nbPoints; p++)
235  {
236  auto xpj = NOMAD::Double(X->get(p,j));
237  lb = lb.isDefined() ? NOMAD::min(lb, xpj) : xpj;
238  ub = ub.isDefined() ? NOMAD::max(ub, xpj) : xpj;
239  }
240  _modelLowerBound[j] = lb;
241  _modelUpperBound[j] = ub;
242  }
243 
244 
245 } // end setModelBounds

276 NOMAD::ArrayOfDouble NOMAD::SgtelibModel::getExtendedLowerBound() const
277 {
278  auto extLowerBound = _pbParams->getAttributeValue<NOMAD::ArrayOfDouble>("LOWER_BOUND");
279 
280  for (size_t i = 0; i < extLowerBound.size(); i++)
281  {
282  if (!extLowerBound[i].isDefined() && _modelLowerBound[i].isDefined() && _modelUpperBound[i].isDefined())
283  {
284  extLowerBound[i] = _modelLowerBound[i]
285  - max(NOMAD::Double(10.0), _modelUpperBound[i] - _modelLowerBound[i]);
286  }
287  }
288 
289  return extLowerBound;

312 {
313  // Manages initialization among other things.
314  NOMAD::Algorithm::startImp();

324 bool NOMAD::SgtelibModel::runImp()
325 {
326  size_t k = 0; // Iteration number
327 
328  if (!_termination->terminate(k))
329  {
330  // Barrier constructor automatically finds the best points in the cache.
331  // This barrier is not the same as the _barrierForX0s member, which
332  // is used for model optimization.
333  // This barrier is used for MegaIteration management.
334  auto barrier = _initialization->getBarrier();
335  if (nullptr == barrier)
336  {
337  auto hMax = _runParams->getAttributeValue<NOMAD::Double>("H_MAX_0");
338  barrier = std::make_shared<NOMAD::Barrier>(hMax, NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this),
339  NOMAD::EvalType::BB,
340  NOMAD::EvcInterface::getEvaluatorControl()->getComputeType());
341  }
342  NOMAD::SuccessType megaIterSuccess = NOMAD::SuccessType::NOT_EVALUATED;
343 
344  while (!_termination->terminate(k))
345  {
346  // Create an MegaIteration: manage multiple iterations at the same time.
347  NOMAD::SgtelibModelMegaIteration megaIteration(this, k, barrier, megaIterSuccess);
348  megaIteration.start();
349  megaIteration.run();
350  megaIteration.end();
351 
352  // Remember these values to construct the next MegaIteration.
353  k = megaIteration.getK();
354  barrier = megaIteration.getBarrier();
355  megaIterSuccess = megaIteration.NOMAD::MegaIteration::getSuccessType();
356 
357  if (_userInterrupt)
358  {
359  hotRestartOnUserInterrupt();
360  }
361  }
362 
363 
364  // member _megaIteration is used for hot restart (read and write)
365  // Update it here.
366  _megaIteration = std::make_shared<NOMAD::SgtelibModelMegaIteration>(this, k, barrier, megaIterSuccess);
367  }
368 
369 
370  _termination->start();
371  _termination->run();
372  _termination->end();
373 
374  NOMAD::OutputQueue::Flush();
375 
376  return true;
377 }

442 size_t NOMAD::SgtelibModel::getNbModels(const NOMAD::SgtelibModelFeasibilityType modelFeasibility,
443  const size_t nbConstraints)
444 {
445  size_t nbModels = -1;
446 
447  switch(modelFeasibility)
448  {
449  case NOMAD::SgtelibModelFeasibilityType::C:
450  nbModels = 1 + nbConstraints;
451  break;
452  case NOMAD::SgtelibModelFeasibilityType::H:
453  case NOMAD::SgtelibModelFeasibilityType::B:
454  case NOMAD::SgtelibModelFeasibilityType::M:
455  nbModels = 2;
456  break;
457  case NOMAD::SgtelibModelFeasibilityType::UNDEFINED:
458  std::cerr << "UNDEFINED SGTELIB_MODEL_FEASIBILITY" << std::endl;
459  break;
460  }
461 
462  return nbModels;
463 }

467 NOMAD::EvalPointSet NOMAD::SgtelibModel::createOraclePoints()
468 {
469  // Create one MegaIteration. It will not be run. It is used to
470  // generate oracle points.
471  // For this reason, k and success are irrelevant.
472  // Barrier points are used to create Iterations. A sub Mads will be run
473  // on every Iteration, using model evaluation.
474  NOMAD::SgtelibModelMegaIteration megaIteration(this, 0, _barrierForX0s, NOMAD::SuccessType::NOT_EVALUATED);
475  megaIteration.generateTrialPoints();
476 
477  NOMAD::OutputQueue::Flush();
478 
479  // The returned EvalPoints are not evaluated by the blackbox, but they will be soon.
480  return megaIteration.getTrialPoints();
481 }
